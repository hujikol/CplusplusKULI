#include <iostream>
using namespace std;
#include <cstdlib>
#include <ctime>
#include <bits/stdc++.h>

int main(){
	printf("\n\n");
	printf("*******************************\n");
	printf("** Random ASCII Generator\n");
	printf("*******************************");
	
	int n = 100;
	int min = 65;//65
	int max = 122;//122
	printf("\n\nGenerating %u random numbers between %u and %u...", n, min, max);
	
	int *numbers = new int[n];
	for(int i = 0; i < n; i++)
		numbers[i] = 0;
	
	
	srand((unsigned)time(0)); 
	for(int i = 0; i < n; i++){
		int r = (rand()%(max-min)) + min;
		numbers[i] = r;
	}
	
	printf(" success.");
	
	printf("\n\nRandom numbers found:");
	for(int i = 0; i < n; i++){
		printf("\n\t%u:\t%c", i+1, numbers[i]);
	}
	printf("\n\n");
	
	
	delete numbers;
	
	return 1;
}
